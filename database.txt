CREATE DATABASE gitpod_db;

USE gitpod_db;

CREATE TABLE Contact (
	contactId INT AUTO_INCREMENT PRIMARY KEY,
	firstName VARCHAR(255) NOT NULL,
	lastName VARCHAR(255) NOT NULL,
	email VARCHAR(255) NOT NULL,
	age INT(3) NOT NULL,
	salary INT NOT NULL,
	address VARCHAR(255) NOT NULL
);
